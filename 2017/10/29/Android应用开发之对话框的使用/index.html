Android应用开发之对话框的使用

发表于 2017-10-29 | 分类于 Android应用开发

MainActivity.java

package com.m520it.myapp;

import java.util.Timer;
import java.util.TimerTask;

import android.app.Activity;
import android.app.AlertDialog;
import android.app.ProgressDialog;
import android.content.DialogInterface;
import android.os.Bundle;
import android.view.View;
import android.view.View.OnClickListener;
import android.widget.Button;
import android.widget.Toast;

public class MainActivity extends Activity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        Button dialogBtn0 = (Button) findViewById(R.id.dialog_btn0);
        Button dialogBtn1 = (Button) findViewById(R.id.dialog_btn1);
        Button dialogBtn2 = (Button) findViewById(R.id.dialog_btn2);
        Button dialogBtn3 = (Button) findViewById(R.id.dialog_btn3);
//
//普通的对话框
//*******************************************************************************************************        
        dialogBtn0.setOnClickListener(new OnClickListener() {

            @Override
            public void onClick(View v) {
                //这里的context一定要用MainActivity，不能用getApplicationContext()方法获取context，这个因为前者MainActivity
                //的生命周期明显要小于整个应用的context的生命周期，而dialog明显是在MainActivity里面的，如果写反的话就出bug了，这里
                //我是深有体会的
                //真的是非常喜欢建造模式的风格
                AlertDialog.Builder builder = new AlertDialog.Builder(MainActivity.this);

                //建造模式创建AlertDialog实例（链式调用）
                AlertDialog dialog = builder.setTitle("提示")   //设置标题
                                            .setMessage("是否退出应用？")  //设置显示文本
                                            .setPositiveButton("是", new DialogInterface.OnClickListener(){

                                                @Override
                                                public void onClick(
                                                        DialogInterface dialog,
                                                        int which) {

                                                    //关闭当前程序
                                                    MainActivity.this.finish();
                                                }

                                            })     //设置确定按钮
                                            .setNegativeButton("否", null)  //设置取消按钮
                                            .create();
                dialog.show();    //这个跟toast差不多，都要show一下
            }

        });
//*******************************************************************************************************        
//
//带有单选的对话框       
//*******************************************************************************************************

        dialogBtn1.setOnClickListener(new OnClickListener() {

            String[] items = new String[]{"苹果","香蕉"};
            int index = 0;

            @Override
            public void onClick(View v) {

                AlertDialog.Builder builder = new AlertDialog.Builder(MainActivity.this);
                AlertDialog dialog = builder.setTitle("请选择水果")
                                             .setSingleChoiceItems(items, 0
                                                     , new DialogInterface.OnClickListener() {

                                                        @Override
                                                        public void onClick(
                                                                DialogInterface dialog,
                                                                int which) {
//                                                            Toast.makeText(MainActivity.this, 
//                                                                    which+"", Toast.LENGTH_SHORT)
//                                                                    .show();
                                                            index = which;
                                                        }

                                             })
                                             .setPositiveButton("确认", 
                                                     new DialogInterface.OnClickListener() {

                                                        @Override
                                                        public void onClick(
                                                                DialogInterface dialog,
                                                                int which) {
                                                            dialog.dismiss();   //退出对话框
                                                            switch (index) {
                                                            case 0:
                                                                Toast.makeText(MainActivity.this, 
                                                                        "选择"+items[index], 
                                                                        Toast.LENGTH_SHORT).show();
                                                                break;

                                                            case 1:
                                                                Toast.makeText(MainActivity.this, 
                                                                        "选择"+items[index], 
                                                                        Toast.LENGTH_SHORT).show();

                                                            }
                                                        }

                                             })
                                             .create();

                dialog.show();

            }

        });

//*******************************************************************************************************
//
//带有多选的对话框
//*******************************************************************************************************

        dialogBtn2.setOnClickListener(new OnClickListener() {

            String[] items = new String[]{"苹果","香蕉","橘子"};
            boolean[] checkedItems = new boolean[]{true,false,false};

            @Override
            public void onClick(View v) {
                AlertDialog.Builder builder = new AlertDialog.Builder(MainActivity.this);
                AlertDialog dialog = builder.setTitle("请选择多个水果")
                                            .setMultiChoiceItems(items, checkedItems, 
                                                    new DialogInterface.OnMultiChoiceClickListener() {

                                                        @Override
                                                        public void onClick(
                                                                DialogInterface dialog,
                                                                int which,
                                                                boolean isChecked) {
                                                            checkedItems[which] = isChecked;
                                                        }

                                                    })
                                             .setPositiveButton("确定", new DialogInterface.OnClickListener() {

                                                @Override
                                                public void onClick(
                                                        DialogInterface dialog,
                                                        int which) {

                                                    String result = "选择的是:";
                                                    for (int i = 0; i < checkedItems.length; i++) {
                                                        if (checkedItems[i]) {
                                                            result += items[i]+"   ";
                                                        }
                                                    }
                                                    //当什么也没选时，进行另外的处理，在项目中一定要多方面的考虑
                                                    if (result.equals("选择的是:")) {
                                                        result = "什么也没选";
                                                    }
                                                    Toast.makeText(MainActivity.this, 
                                                            result, Toast.LENGTH_SHORT).show();

                                                    dialog.dismiss();
                                                }

                                             })
                                             .create();

                dialog.show();                                

            }

        });

//*******************************************************************************************************
//
//带有进度条的对话框，这里没有使用建造模式，是直接创建了一个进度条对话框，因为这个进度条对话框本身需要初始化的参数可能比较少，不使用建造模式也能看得很清爽
//******************************************************************************************************* 

        dialogBtn3.setOnClickListener(new OnClickListener() {

            @Override
            public void onClick(View v) {

                final ProgressDialog dialog = new ProgressDialog(MainActivity.this);
                dialog.setTitle("提示");
                dialog.setMessage("加载中...");
                dialog.show();

                //在上面的对话框会一直转转转，我们希望能设置一个定时器，在特定的时间之后自动关闭这个进度条对话框
                //现在设置在3秒之后关闭对话框
                //实际上定时器也是一个子线程，当然如果更改主线程的控件，则需要使用runOnUiThread方法
                new Timer().schedule(new TimerTask() {

                    public void run() {

                        //下面的代码是将run方法里面包含的代码放到主线程中去运行
                        runOnUiThread(new Runnable() {

                            @Override
                            public void run() {

                                dialog.dismiss();

                            }
                        });

                    }

                }, 3000);

            }

        });

//*******************************************************************************************************

    }

}
