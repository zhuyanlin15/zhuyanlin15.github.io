package com.m520it.test;

import android.app.Activity;
import android.os.Bundle;
import android.widget.CheckBox;
import android.widget.CompoundButton;
import android.widget.RadioGroup;
import android.widget.Toast;


public class MainActivity extends Activity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        //1、获取RadioGroup的id号
        RadioGroup radioGroup = (RadioGroup) findViewById(R.id.radioGroup);

        //2、设置RadioGroup的事件
        radioGroup.setOnCheckedChangeListener(new RadioGroup.OnCheckedChangeListener() {

            @Override
            public void onCheckedChanged(RadioGroup group, int checkedId) {
                //3、判断点击的RadioButton
                if(checkedId == R.id.radioButton1) {
                    Toast.makeText(getApplicationContext(), "男", Toast.LENGTH_SHORT).show();
                }
                else if(checkedId == R.id.radioButton2) {
                    Toast.makeText(getApplicationContext(), "女", Toast.LENGTH_SHORT).show();
                }
            }
        });

        CheckBox appleCbx = (CheckBox) findViewById(R.id.apple_cbx);
        //一定要加上CompoundButton，因为上面已经有了RadioButton的OnCheckChangeListener了
        appleCbx.setOnCheckedChangeListener(new  CompoundButton.OnCheckedChangeListener() {

            @Override
            public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
                Toast.makeText(getApplicationContext(), isChecked?"选中苹果":"未选中苹果", Toast.LENGTH_SHORT).show();
            }

        });

        CheckBox bananaCbx = (CheckBox) findViewById(R.id.banana_cbx);
        //一定要加上CompoundButton，因为上面已经有了RadioButton的OnCheckChangeListener了
        bananaCbx.setOnCheckedChangeListener(new  CompoundButton.OnCheckedChangeListener() {

            @Override
            public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
                Toast.makeText(getApplicationContext(), isChecked?"选中香蕉":"未选中香蕉", Toast.LENGTH_SHORT).show();
            }

        });

        CheckBox orangeCbx = (CheckBox) findViewById(R.id.orange_cbx);
        //一定要加上CompoundButton，因为上面已经有了RadioButton的OnCheckChangeListener了
        orangeCbx.setOnCheckedChangeListener(new  CompoundButton.OnCheckedChangeListener() {

            @Override
            public void onCheckedChanged(CompoundButton buttonView, boolean isChecked) {
                Toast.makeText(getApplicationContext(), isChecked?"选中橘子":"未选中橘子", Toast.LENGTH_SHORT).show();
            }

        });
    }
}

MainActivity

package com.m520it.temp;

import android.annotation.SuppressLint;
import android.app.Activity;
import android.os.Bundle;
import android.webkit.JsResult;
import android.webkit.WebChromeClient;
import android.webkit.WebView;
import android.webkit.WebViewClient;


public class MainActivity extends Activity {

    @SuppressLint("SetJavaScriptEnabled") @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        //1、找到WebView控件
        WebView webView = (WebView) findViewById(R.id.webView1);

        //2、加载网页
        webView.loadUrl("http://www.qq.com");

        webView.setWebViewClient(new WebViewClient() {

            @Override
            public boolean shouldOverrideUrlLoading(WebView view, String url) {

                //3、在当前的WebView中显示
                view.loadUrl(url);
                //4、禁止网页跳转
                return true;
            }
        });

        //5、JS默认是关闭的，可以打开
        webView.getSettings().setJavaScriptEnabled(true);

        //6、JS的弹窗如alerm()，prompt()，confirm()···默认是关闭的，而HTML的弹窗是默认开启的，当然是可以打开JS的弹窗的
        webView.setWebChromeClient(new WebChromeClient() {

            @Override
            public boolean onJsAlert(WebView view, String url, String message,
                    JsResult result) {
                return false;
            }

        });
    }
}
