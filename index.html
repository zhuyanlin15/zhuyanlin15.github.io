s3c2440嵌入式之SDRAM

发表于 2017-09-22

#代码

##head.s
.equ sdram_base, 0x30000000 @仅仅是宏定义而已
.equ mem_ctrl_reg, 0x48000000 

@;////////////////////////////////////////////////////////////////////////////////////////////    
.text
.global _start
_start:
    bl close_watch_dog
    bl set_memort_control_register
    bl steppingstone_copy_sdram
    ldr pc,=sdram_on
sdram_on:                                @如果这样ldr pc,=sdram_base，则cpu会一直从0x30000000运行，陷入死循环
    ldr sp,=0x34000000                    @一定别忘记设置栈
    bl main
main_loop:
    b main_loop

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @1、关闭看门狗
close_watch_dog:
    ldr r0,=0x53000000
    mov r1,#0x0000
    str r1,[r0]
    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @2、设置13个存储控制器，配置从外部SRAM中启动
set_memort_control_register:
    ldr r0,=mem_ctrl_reg
    adrl r1,memory_control_register_value
    add r3,r0,#52

loop1:
    ldr r2,[r1],#4
    str    r2,[r0],#4
    cmp r0,r3
    bne    loop1

    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
                                        @3、将steppingstone中的代码移动到外部SRAM中
steppingstone_copy_sdram:
    mov r1,#0
    ldr r2,=sdram_base
    add r3,r1,#1024*4
loop2:
    ldr r4,[r1],#4
    str r4,[r2],#4
    cmp r1,r3
    bne loop2

    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @存储控制器的13个寄存器的值
.align 4   @好像少了这个是不行的
memory_control_register_value:
    .long    0x22011110    @BWSCON
    .long    0x00000700    @BANKCON0
    .long    0x00000700    @BANKCON1
    .long    0x00000700    @BANKCON2
    .long    0x00000700    @BANKCON3
    .long    0x00000700    @BANKCON4
    .long    0x00000700    @BANKCON5
    .long    0x00018005    @BANKCON6
    .long    0x00018005    @BANKCON7
    .long    0x008C07A3    @REFRESH
    .long    0x000000B1    @BANKSIZE
    .long    0x00000030    @MRAR6
    .long    0x00000030    @MRAR7

##leds.c

#define GPFCON    *(volatile unsigned int*)0x56000050
#define GPFDAT    *(volatile unsigned int*)0x56000054

#define GPF4CON_mask    0x0300
#define GPF5CON_mask    0x0C00
#define GPF6CON_mask    0x3000

#define GPF4_out    0x0100
#define GPF5_out    0x0400
#define GPF6_out    0x1000

void delay(unsigned int t)
{
    while(t--);
}

int main(void)
{
    short i = 4;
    GPFCON &= (GPF4CON_mask | GPF5CON_mask | GPF6CON_mask);
    GPFCON |= (GPF4_out | GPF5_out | GPF6_out);
    GPFDAT &= (7<<4);
    while(1)
    {
        GPFDAT ^= (1<<i++);
        delay(10000);
        if(i == 7)
        {
            i = 4;
        }
    }
    return 0;
}
