package com.m520it.post;

import java.io.BufferedReader;
import java.io.InputStream;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

import android.app.Activity;
import android.os.Bundle;
import android.view.View;
import android.view.View.OnClickListener;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;

import com.m520it.temp.R;

public class MainActivity extends Activity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        //获取控件
        Button loginBtn = (Button) findViewById(R.id.login_btn);
        final EditText usernameEt = (EditText) findViewById(R.id.username_et);
        final EditText passwordEt = (EditText) findViewById(R.id.password_et);


        loginBtn.setOnClickListener(new OnClickListener() {

            @Override
            public void onClick(View v) {

                    new Thread() {

                        final String username = usernameEt.getText().toString();
                        final String password = passwordEt.getText().toString();

                        public void run() {

                            try {
                                //点击登录时访问的地址
                                String UrlPath = "http://192.168.176.1:8080/loginPost/login";

                                URL url = new URL(UrlPath);
                                HttpURLConnection openConnection = (HttpURLConnection) url.openConnection();
                                openConnection.setRequestMethod("POST");

                                /**************这里的代码是POST和GET最大的不同点**************/
                                //告诉服务器，我们是一个form表单的提交，组成为属性加上内容
                                String info = "username="+username+"&pwd="+password;
                                openConnection.setRequestProperty("Content-Type", "application/x-www-form-urlencoded");
                                openConnection.setRequestProperty("Content-Length", info.length()+"");
                                //使用输出流的方式提交，视频里面老师说Http的输出流默认是关闭的，但是我没使用setDoOutput方法打开也能成功提交
                                openConnection.setDoOutput(true);
                                openConnection.getOutputStream().write(info.getBytes());
                                /****************************************************/

                                if (openConnection.getResponseCode() == 200) {
                                    InputStream inputStream = openConnection.getInputStream();

                                    //因为读取的是字节流，而我们要显示字符串，则需要将其转换为字符流
                                    BufferedReader reader = new BufferedReader(new InputStreamReader(inputStream));

                                    //匿名内部类的变量是要加上final来修饰的
                                    final String readLine = reader.readLine();

                                    //将显示在UI里的代码放到主线程中
                                    runOnUiThread(new Runnable() {
                                        public void run() {
                                            Toast.makeText(MainActivity.this,readLine,Toast.LENGTH_SHORT).show();
                                        };
                                    });
                                }
                            } catch (Exception e) {
                                e.printStackTrace();
                            }

                        }

                    }.start();

            }

        });
    }

}

package com.m520it.newsclient.adapter;

import java.util.ArrayList;

import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.BaseAdapter;
import android.widget.TextView;

import com.m520it.newsclient.bean.NewsBean;
import com.m520it.newsclient.ui.ImageViewWithUrl;
import com.m520it.temp.R;

public class MyAdapter extends BaseAdapter {

    private ArrayList<NewsBean> mNewsList;

    //通过这个方法和ArrayList建立连接
    public void setDatas(ArrayList<NewsBean> mNewsInfo) {
        mNewsList = mNewsInfo;
    }

    //得到ArrayList中的数量
    @Override
    public int getCount() {
        return mNewsList != null?mNewsList.size():0;
    }

    class Holder {
        ImageViewWithUrl newsImgIv;
        TextView newsTitleTv;
        TextView newsDiscTv;
        TextView newsTypeTv;
    }

    //可以将一组数据传到ListView中
    @Override
    public View getView(int position, View convertView, ViewGroup parent) {
        Holder newsHolder = null;
        if (convertView == null) {
            LayoutInflater inflater = LayoutInflater.from(parent.getContext());
            convertView = inflater.inflate(R.layout.news_layout, null);
            //这里做一个判断，当convertView为空的时候，则创建一个newsHolder对象，用于存放拿到的每个列表中的控件
            //之所以有个判断语句是防止每次完成一个列表时，又重新创建一个newsHolder对象，这样提高了笑效率
            newsHolder = new Holder();
            newsHolder.newsImgIv = (ImageViewWithUrl) convertView.findViewById(R.id.img_iv);
            newsHolder.newsTitleTv = (TextView) convertView.findViewById(R.id.title_tv);
            newsHolder.newsDiscTv = (TextView) convertView.findViewById(R.id.disc_tv);
            newsHolder.newsTypeTv = (TextView) convertView.findViewById(R.id.type_tv);

            convertView.setTag(newsHolder);
        }else {
            newsHolder = (Holder) convertView.getTag();
        }

        //既然获取了信息，则将各种信息使用刚刚获取的控件表达出来
        NewsBean newsBean = mNewsList.get(position);
        newsHolder.newsImgIv.setImageUrl(newsBean.getImage());
        newsHolder.newsTitleTv.setText(newsBean.getTitle());
        newsHolder.newsDiscTv.setText(newsBean.getDescription());

        //判断是评论、专题还是科技类型
        initType(newsHolder, newsBean);
        return convertView;
    }

    private void initType(Holder newsHolder, NewsBean newsBean) {
        if (newsBean.getType() == 1) {
            newsHolder.newsTypeTv.setText("评论："+newsBean.getComment());
        }else if (newsBean.getType() == 2) {
            newsHolder.newsTypeTv.setText("专题");
        }else if (newsBean.getType() == 3) {
            newsHolder.newsTypeTv.setText("科技");
        }
    }

    @Override
    public Object getItem(int position) {
        return null;
    }

    @Override
    public long getItemId(int position) {
        return 0;
    }

}

package com.m520it.newsclient.ui;

import java.io.InputStream;
import java.net.HttpURLConnection;
import java.net.URL;

import android.annotation.SuppressLint;
import android.content.Context;
import android.graphics.Bitmap;
import android.graphics.BitmapFactory;
import android.os.Handler;
import android.os.Message;
import android.util.AttributeSet;
import android.widget.ImageView;

import com.m520it.temp.R;

@SuppressLint("HandlerLeak")
public class ImageViewWithUrl extends ImageView {

    public static final int GET_IMAGE_URL_SUCCESS = 0x01;
    public static final int GET_IMAGE_URL_FAILED = 0x02;

    //定义一个Handler对象用来接收子线程发送的消息并处理
    Handler mHandler = new Handler() {
        public void handleMessage(Message msg) {

            //分为图片加载成功和图片加载失败
            if (msg.what == GET_IMAGE_URL_SUCCESS) {

                setImageBitmap((Bitmap) msg.obj);

            } else if (msg.what == GET_IMAGE_URL_FAILED) {
                setImageResource(R.drawable.ic_launcher);
            }
        };
    };

    // new一个控件的时候调用
    public ImageViewWithUrl(Context context) {
        super(context);
    }

    // 这个和xml文件的ui有关系
    public ImageViewWithUrl(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    /**
     * 注意啊，网络请求一定要放在子线程里面，改变ui一定要在UI线程（主线程）里面，在本程序中使用的Handler处理子线程
     */
    public void setImageUrl(final String imageUrl) {
        new Thread() {
            public void run() {

                try {
                    URL url = new URL(imageUrl);
                    HttpURLConnection httpOpenConnection = (HttpURLConnection) url
                            .openConnection();
                    if (httpOpenConnection.getResponseCode() == 200) {
                        InputStream inputStream = httpOpenConnection.getInputStream();
                        Bitmap bitmap = BitmapFactory.decodeStream(inputStream);
                        Message msg = new Message();

                        //一个int类型的条件，用于主线程判断是加载成功还是失败
                        msg.what = GET_IMAGE_URL_SUCCESS;
                        msg.obj = bitmap;
                        mHandler.sendMessage(msg);
                    } else {
                        Message msg = new Message();
                        msg.what = GET_IMAGE_URL_FAILED;
                        mHandler.sendMessage(msg);
                    }
                } catch (Exception e) {
                    e.printStackTrace();

                    //出现异常当然也是加载失败了
                    Message msg = new Message();
                    msg.what = GET_IMAGE_URL_FAILED;
                    mHandler.sendMessage(msg);
                }
            }

        }.start();

    }

}

//这个工具类的功能是获取xml文件，并将其保存为流
public class NetworkUtil {

    //帮助类的方法当然是static了
    public static InputStream getResource(String string) throws Exception {
        URL url = new URL(string);

        HttpURLConnection openConnection = (HttpURLConnection) url.openConnection();

        int responseCode = openConnection.getResponseCode();

        if (responseCode == 200) {

            return openConnection.getInputStream();
        }

        return null;
    }

com.m520it.newsclient.util.XmlUtil.java

package com.m520it.newsclient.util;

import java.io.InputStream;
import java.util.ArrayList;

import org.xmlpull.v1.XmlPullParser;

import android.util.Xml;

import com.m520it.newsclient.bean.NewsBean;

public class XmlUtil {

    public static ArrayList<NewsBean> parseXml(InputStream is) throws Exception {

        ArrayList<NewsBean> returnVal = new ArrayList<NewsBean>();

        //创建一个xml的pull解析器，返回类型是XmlPullParser类
        XmlPullParser newPullParser = Xml.newPullParser();

        //设置输入流并设置编码方式
        newPullParser.setInput(is, "utf-8");

        //获取xml文件的标签
        int eventType = newPullParser.getEventType();
        NewsBean newsBean = null;

        //一直读取标签，直到xml文件的结束
        while (eventType != XmlPullParser.END_DOCUMENT) {

            //如果是开始标签，则判断标签的类型并作出相应的处理
            if (eventType == XmlPullParser.START_TAG) {
                if (newPullParser.getName().equals("item")) {
                    newsBean = new NewsBean();

                }else if (newPullParser.getName().equals("title")) {
                    newsBean.setTitle(newPullParser.nextText());

                }else if (newPullParser.getName().equals("description")) {
                    newsBean.setDescription(newPullParser.nextText());

                }else if (newPullParser.getName().equals("image")) {
                    newsBean.setImage(newPullParser.nextText());

                }else if (newPullParser.getName().equals("type")) {
                    newsBean.setType(Integer.parseInt(newPullParser.nextText()));

                }else if (newPullParser.getName().equals("comment")) {
                    newsBean.setComment(Integer.parseInt(newPullParser.nextText()));

                }

            //当读到结束标签时，则保存到Arraylist类型对象中
            }else if (eventType == XmlPullParser.END_TAG) {
                if (newPullParser.getName().equals("item")) {
                    returnVal.add(newsBean);
                }
            }

            //一直往下查询，否则就死循环了
            eventType = newPullParser.next();
        }

        return returnVal;
    }

}

Android开发模块之从网络获取图片资源）（多删少补）

发表于 2017-10-22 | 分类于 Android开发模块

模块介绍

本模块为基础模块，其功能是：从给定的网站获取图片资源，并将其放到输入流中，等待后续操作。

分析

1)首先创建一个URL类，并将网址作为输入参数

URL url = new URL("https://www.baidu.com");

2)打开URL的连接

HttpURLConnection ocnn = (HttpURLConnection) url.openConnection();

注意：openConnection()的返回类型是URLConnection类，但是因为要使用Http协议，所以最好进行强制转换为HttpURLConnection类，否则后面就不能获取访问状态码

3)判断ocnn的访问状态码，判断是否为200。当为200时，访问成功，执行后续的操作，否则访问失败。

int responseCode = ocnn.getResponseCode(); 
if (responseCode == 200) {
    //后续操作
}

注意：为了使代码更加健壮，可以对访问失败进行处理

4)使用getInputStream()方法获取连接的资源并转换成输入流

InputStream inputStream = ocnn.getInputStream();

重点

1)在安卓高版本中规定网络访问一定要在子线程中，防止网络出现错误后影响主线程；

2)记得要加上try-catch或者抛出异常。

package com.m520it.sqlite3;

import android.content.ContentValues;
import android.content.Context;

/**
 *    实现增删查改
 */
public class ContactDao {

    private DbOpenHelper mHelper;

    public ContactDao(Context c) {
        mHelper=new DbOpenHelper(c);
    }

    /**
     * 插入一条联系人信息
     * @param username
     * @param phone
     */
    public boolean insertContact(String username,String phone){
        SQLiteDatabase db = mHelper.getWritableDatabase();
        // _id username phone
        //nullColumnHack 比如 放了username 那么插入的时候 username就不能为null
        //如果nullColumnHack设置为null  那么此次插入的字段可以允许某些字段不插入
        //ContentValues 键值对 源代码里面只是简单的维护了一个HashMap
        ContentValues values=new ContentValues();
        //key 插入的数据库表的字段名  value就是要插入的值
        values.put(DbOpenHelper._USERNAME, username);
        values.put(DbOpenHelper._PHONE, phone);
        //@return 新插入的行的id 如果是-1则说明插入失败
        long rowId = db.insert(DbOpenHelper.TABLE_NAME, null , values);
        return rowId!=-1;
    }

    public boolean updateContact(String username,String newPhone){
        SQLiteDatabase db = mHelper.getWritableDatabase();
        ContentValues values=new ContentValues();
        values.put(DbOpenHelper._PHONE, newPhone);
        //whereClause  where语句
        //whereArgs where语句的绑定值
        //@return int 更新了几行就返回多少
        int updateRows = db.update(DbOpenHelper.TABLE_NAME,
                values, DbOpenHelper._USERNAME+"=?", new String[]{username});
        return updateRows>0;
    }

    public boolean deleteContact(String username){
        SQLiteDatabase db = mHelper.getWritableDatabase();
        //whereClause  where语句
        //whereArgs where语句的绑定值
        int deleteRows = db.delete(DbOpenHelper.TABLE_NAME, "username=?", new String[]{username});
        return deleteRows>0;
    }

    public String queryContact(String searchPhone){

        String result="";

        SQLiteDatabase db = mHelper.getReadableDatabase();
//        columns String[] 返回的行
        Cursor cursor = db.query(DbOpenHelper.TABLE_NAME, 
                new String[]{DbOpenHelper._USERNAME,DbOpenHelper._PHONE},
                "phone=?", new String[]{searchPhone}, 
                null, null, null);
        while (cursor.moveToNext()) {
            //传入一个列的列名 返回一个列的索引
            int columnIndex = cursor.getColumnIndex(DbOpenHelper._USERNAME);
            //传入一个列的索引 返回该列下 某一行的值
            String username = cursor.getString(columnIndex);

            //传入一个列的列名 返回一个列的索引
            int phoneColumnIndex = cursor.getColumnIndex(DbOpenHelper._PHONE);
            //传入一个列的索引 返回该列下 某一行的值
            String phone = cursor.getString(phoneColumnIndex);

            result+=(username+"  "+phone+" \n");
        }
        return result;
    }


}

package com.m520it.saveandparseXML;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileWriter;
import java.io.OutputStream;

import org.xmlpull.v1.XmlPullParser;
import org.xmlpull.v1.XmlSerializer;

import android.content.Context;
import android.util.Xml;

public class XMLUtil {

    /**
     * 以此方法保存的xml文件效果：
     * <map>
     * <string name="usename">zhangsan</string>
     * <string name="psk">123456</string>
     * </map>
     * 一定有异常用throw，可能存在异常用throws
     */
    public static void saveXMLByStringBuilder(MainActivity mainActivity,
            String passport, String password) throws Exception {

        //1、建立一个字符串，相比于String，StringBuilder是可变的
        StringBuilder stringBuilder = new StringBuilder();
        stringBuilder.append("<map>");
        stringBuilder.append("<string name=\"usename\">"+passport+"</string>");
        stringBuilder.append("<string name=\"psk\">"+password+"</string>");
        stringBuilder.append("</map>");
        String result = stringBuilder.toString();

        //2、创建文件的句柄，getFileDir()是在Content类下的，而MainActivity是继承自Content
        File f = new File(mainActivity.getFilesDir(),"qqinfo.xml");

        //3、使用字符流，并且要进行可能存在异常的检查，我们可以将方法的异常抛出
        BufferedWriter bw = new BufferedWriter(new FileWriter(f));

        //4、写入刚刚建立的字符串
        bw.write(result);

        //5、初次之外，还需进行刷新，当然关闭文件也是可以的，里面已经进行了刷新
        //bw.flush();
        bw.close();
    }

    /**
     * 以此方法保存的xml文件效果：
     * <map>
     * <username>zhangsan</username>
     * <psk>123456</psk>
     * </map>
     */
    public static void saveXMLByXMLSerializer(MainActivity mainActivity,
            String passport, String password) throws Exception {

         //1、定义一个xml序列化器
         XmlSerializer xmlSerializer = Xml.newSerializer();

         //2、需要一个输出流，使用Content中的openFileOutput()比较方便
         OutputStream os = mainActivity.openFileOutput("qqinfo1.xml", Context.MODE_PRIVATE);

         //3、设置输出路径和格式
         xmlSerializer.setOutput(os, "utf-8");

         //4、设置xml文件的开始、各个开标签闭标签和标签里面的内容，但是把标签属性给去掉了
         xmlSerializer.startDocument("utf-8", true);

         xmlSerializer.startTag(null, "map");

         xmlSerializer.startTag(null, "username");
         xmlSerializer.text(passport);
         xmlSerializer.endTag(null, "username");

         xmlSerializer.startTag(null, "psk");
         xmlSerializer.text(password);
         xmlSerializer.endTag(null, "psk");

         xmlSerializer.endTag(null, "map");

         xmlSerializer.endDocument();

    }

    public static String parseXMLByPullParser(MainActivity mainActivity) throws Exception {

        String string = "";

        //1、创建一个PULL解析器
        XmlPullParser xmlPullParser = Xml.newPullParser();

        //2、使用流告诉解析器文件的位置和编码
        FileInputStream fis = mainActivity.openFileInput("qqinfo1.xml");
        xmlPullParser.setInput(fis, "utf-8");

        int eventType = xmlPullParser.getEventType();
//        这里要将整型转换成字符串型
//        Log.v("520it",eventType + "");
//        
//        eventType = xmlPullParser.next();
//        Log.v("520it",eventType + "");
//        Log.v("520it",xmlPullParser.getName());
//        
//        eventType = xmlPullParser.next();
//        Log.v("520it",eventType + "");
//        Log.v("520it",xmlPullParser.getName());

        //3、查询状态，并取出内容，一直到文档的结束
        while(eventType != XmlPullParser.END_DOCUMENT) {
            if(eventType == XmlPullParser.START_TAG) {
                if(xmlPullParser.getName().equals("username")) {
                    string += "username = " + xmlPullParser.nextText() + "\n";
                }else if (xmlPullParser.getName().equals("psk")){
                    string += "psk = " + xmlPullParser.nextText() + "\n";
                }
            }
            eventType = xmlPullParser.next();
        }

        return string;
    }
}

XMLUtil.java

package com.m520it.SaveXML;

import java.io.BufferedWriter;
import java.io.File;
import java.io.FileWriter;

public class XMLUtil {

    //一定有异常用throw，可能存在异常用throws
    public static void saveXMLByStringBuilder(MainActivity m,String passport, String password) throws Exception {

        //1、建立一个字符串，相比于String，StringBuilder是可变的
        StringBuilder stringBuilder = new StringBuilder();
        stringBuilder.append("<map>");
        stringBuilder.append("<string name=\"usename\">"+passport+"</string>");
        stringBuilder.append("<string name=\"psk\">"+password+"</string>");
        stringBuilder.append("</map>");
        String result = stringBuilder.toString();

        //2、创建文件的句柄，getFileDir()是在Content类下的，而MainActivity是继承自Content
        File f = new File(m.getFilesDir(),"qqinfo.xml");

        //3、使用字符流，并且要进行可能存在异常的检查，我们可以将方法的异常抛出
        BufferedWriter bw = new BufferedWriter(new FileWriter(f));

        //4、写入刚刚建立的字符串
        bw.write(result);

        //5、初次之外，还需进行刷新，当然关闭文件也是可以的，里面已经进行了刷新
        //bw.flush();
        bw.close();
    }
}
