S3C2440嵌入式之mkconfig文件分析

发表于 2017-10-11 | 分类于 嵌入式linux开发

mkconfig文件内容（关键语句旁边用“##”注释）

#!/bin/sh -e

# Script to create header files and links to configure
# U-Boot for a specific board.
#
# Parameters:  Target  Architecture  CPU  Board [VENDOR] [SOC]
#
# (C) 2002-2006 DENX Software Engineering, Wolfgang Denk <wd@denx.de>
#

##MakeFile中的配置指令，即执行make 100ask24x0_config
##mkconfig 100ask24x0 arm arm920t 100ask24x0 NULL s3c24x0
##              $1      $2    $3        $4     $5    $6

APPEND=no    # Default: Create new config file
BOARD_NAME=""    # Name to print in make output

while [ $# -gt 0 ] ; do
    case "$1" in
    --) shift ; break ;;
    -a) shift ; APPEND=yes ;;
    -n) shift ; BOARD_NAME="${1%%_config}" ; shift ;;
    *)  break ;;
    esac
done
##上面16~22行是检查指令中是否有--,-a,-n,*的，显然是没有的

[ "${BOARD_NAME}" ] || BOARD_NAME="$1"
##这个和C语言的条件编译好像，实际上这行代码与下一行意思一样
##BOARD_NAME = 100ask24x0

[ $# -lt 4 ] && exit 1
[ $# -gt 6 ] && exit 1
##参数个数小于4个或者大于6个则退出

echo "Configuring for ${BOARD_NAME} board..."
##打印后面的字符串

#
# Create link to architecture specific headers
#
if [ "$SRCTREE" != "$OBJTREE" ] ; then
##显然条件不成立，执行else语句
    mkdir -p ${OBJTREE}/include
    mkdir -p ${OBJTREE}/include2
    cd ${OBJTREE}/include2
    rm -f asm
    ln -s ${SRCTREE}/include/asm-$2 asm
    LNPREFIX="../../include2/asm/"
    cd ../include
    rm -rf asm-$2
    rm -f asm
    mkdir asm-$2
    ln -s asm-$2 asm
else
    cd ./include
    ##此时已经进入到了include文件夹中
    rm -f asm
    ln -s asm-$2 asm
fi

rm -f asm-$2/arch

if [ -z "$6" -o "$6" = "NULL" ] ; then
##判断第6个参数是为空或是等于"NULL"
    ln -s ${LNPREFIX}arch-$3 asm-$2/arch
else
    ln -s ${LNPREFIX}arch-$6 asm-$2/arch
fi

if [ "$2" = "arm" ] ; then
    rm -f asm-$2/proc
    ln -s ${LNPREFIX}proc-armv asm-$2/proc
fi

#
# Create include file for Make
#
echo "ARCH   = $2" >  config.mk
echo "CPU    = $3" >> config.mk
echo "BOARD  = $4" >> config.mk
##一个"<"表示创建文件，两个"<<"表示追加内容
##上面78~81行相当于config.mk内容为
##ARCH = arm
##CPU = arm920t
## BOARD = 100ask24x0

[ "$5" ] && [ "$5" != "NULL" ] && echo "VENDOR = $5" >> config.mk

[ "$6" ] && [ "$6" != "NULL" ] && echo "SOC    = $6" >> config.mk
##继续在config.mk中追加SOC = s3c24x0

#
# Create board specific header file
#
##创建一个单板相关的头文件
if [ "$APPEND" = "yes" ]    # Append to existing config file
then
    echo >> config.h
else
    > config.h        # Create new config file
fi
echo "/* Automatically generated - do not edit */" >>config.h
echo "#include <configs/$1.h>" >>config.h

exit 0

    private EditText mPasswordEt;

    @Override

    public void loginClick(View v) {

        //3、获取EditText中的文本
        String passport = mPassportEt.getText().toString();
        String password = mPasswordEt.getText().toString();

        //4、判断账号和密码是不是为空，调用TextUtils工具类进行判断
        if(TextUtils.isEmpty(passport)&&TextUtils.isEmpty(password)) {
            Toast.makeText(getApplicationContext(), "请填写账号和密码", 0).show();
            //视频教程中说加上return，但是我试了一下，不加return也没有什么影响
            //return;
        }

        else if(TextUtils.isEmpty(passport)) {
            Toast.makeText(getApplicationContext(), "请填写账号", 0).show();
        }

        else if(TextUtils.isEmpty(password)) {
            Toast.makeText(getApplicationContext(), "请填写密码", 0).show();
        }

        //5、保存信息，主要是账号和密码
        savaData(passport,password);

    }

    private void savaData(String passport, String password) {

        try {
            //5-1、声明一个文件类
            //File file = new File("/data/data/com.m520it.temp/cache/qqinfo.txt");
            File file = new File(getCacheDir(), "qqinfo.txt");

            //5-2、声明一个字符流缓冲区，实际上FileWriter类型变量才是写入数据的主力而不是File类型变量
            BufferedWriter writer = new BufferedWriter(new FileWriter(file));

            //5-3、写入字符串，账号和密码以"#"分隔
            writer.write(passport+"#"+password);

            //5-4、关闭文件
            writer.close();
        } catch (IOException e) {
            e.printStackTrace();
        }

    }

    private void readData() {


        try {
            //6-1、声明一个文件类
            //File file = new File("/data/data/com.m520it.temp/cache/qqinfo.txt");
            File file = new File(getCacheDir(), "qqinfo.txt");

            //6-2、检查文件是否存在，为了保险起见，同时检查文件中是否有内容
            if(file.exists()) {

                //6-3、字符流缓冲区，构造器中为FileReader类
                BufferedReader reader = new BufferedReader(new FileReader(file));

                //6-4、读取文件中的单行数据
                String line = reader.readLine();

                //6-5、关闭文件
                reader.close();

                //6-6、判断文件中是否用"#"将账号和密码分隔开来
                if(line.contains("#")) {

                    //6-7、以"#"为参考分隔成两个字符串
                    String[] datas = line.split("#");

                    //6-8、使EditText回显
                    mPassportEt.setText(datas[0]);
                    mPasswordEt.setText(datas[1]);

                }
            }

            //**注意这里原来是FileNotFoundException，但是因为已经检查文件是否存在了，就改为Exception
        } catch (Exception e) {
            e.printStackTrace();
        }

    }

}

S3C2440嵌入式之MMU

发表于 2017-09-28 | 分类于 嵌入式linux开发

代码

head.s

@*************************************************************************
@ File：head.S
@ 功能：设置SDRAM，将第二部分代码复制到SDRAM，设置页表，启动MMU，
@       然后跳到SDRAM继续执行
@*************************************************************************       

.text
.global _start
_start:
    ldr sp, =4096                       @ 设置栈指针，以下都是C函数，调用前需要设好栈
    bl  disable_watch_dog               @ 关闭WATCHDOG，否则CPU会不断重启
    bl  memsetup                        @ 设置存储控制器以使用SDRAM
    bl  copy_2th_to_sdram               @ 将第二部分代码复制到SDRAM
    bl  create_page_table               @ 设置页表
    bl  mmu_init                        @ 启动MMU
    ldr sp, =0xB4000000                 @ 重设栈指针，指向SDRAM顶端(使用虚拟地址)
    ldr pc, =0xB0004000                 @ 跳到SDRAM中继续执行第二部分代码
halt_loop:
    b   halt_loop

init.c

/*
 * init.c: 进行一些初始化，在Steppingstone中运行
 * 它和head.S同属第一部分程序，此时MMU未开启，使用物理地址
 */ 

/* WATCHDOG寄存器 */
#define WTCON           (*(volatile unsigned long *)0x53000000)
/* 存储控制器的寄存器起始地址 */
#define MEM_CTL_BASE    0x48000000


/*
 * 关闭WATCHDOG，否则CPU会不断重启
 */
void disable_watch_dog(void)
{
    WTCON = 0;  // 关闭WATCHDOG很简单，往这个寄存器写0即可
}

/*
 * 设置存储控制器以使用SDRAM
 */
void memsetup(void)
{
    /* SDRAM 13个寄存器的值 */
    unsigned long  const    mem_cfg_val[]={ 0x22011110,     //BWSCON
                                            0x00000700,     //BANKCON0
                                            0x00000700,     //BANKCON1
                                            0x00000700,     //BANKCON2
                                            0x00000700,     //BANKCON3  
                                            0x00000700,     //BANKCON4
                                            0x00000700,     //BANKCON5
                                            0x00018005,     //BANKCON6
                                            0x00018005,     //BANKCON7
                                            0x008C07A3,     //REFRESH
                                            0x000000B1,     //BANKSIZE
                                            0x00000030,     //MRSRB6
                                            0x00000030,     //MRSRB7
                                    };
    int     i = 0;
    volatile unsigned long *p = (volatile unsigned long *)MEM_CTL_BASE;
    for(; i < 13; i++)
        p[i] = mem_cfg_val[i];
}

/*
 * 将第二部分代码复制到SDRAM
 */
void copy_2th_to_sdram(void)
{
    unsigned int *pdwSrc  = (unsigned int *)2048;
    unsigned int *pdwDest = (unsigned int *)0x30004000;

    while (pdwSrc < (unsigned int *)4096)
    {
        *pdwDest = *pdwSrc;
        pdwDest++;
        pdwSrc++;
    }
}

/*
 * 设置页表
 */
void create_page_table(void)
{

/* 
 * 用于段描述符的一些宏定义
 */ 
#define MMU_FULL_ACCESS     (3 << 10)   /* 访问权限 */
#define MMU_DOMAIN          (0 << 5)    /* 属于哪个域 */
#define MMU_SPECIAL         (1 << 4)    /* 必须是1 */
#define MMU_CACHEABLE       (1 << 3)    /* cacheable */
#define MMU_BUFFERABLE      (1 << 2)    /* bufferable */
#define MMU_SECTION         (2)         /* 表示这是段描述符 */
#define MMU_SECDESC         (MMU_FULL_ACCESS | MMU_DOMAIN | MMU_SPECIAL | \
                             MMU_SECTION)
#define MMU_SECDESC_WB      (MMU_FULL_ACCESS | MMU_DOMAIN | MMU_SPECIAL | \
                             MMU_CACHEABLE | MMU_BUFFERABLE | MMU_SECTION)
#define MMU_SECTION_SIZE    0x00100000

    unsigned long virtuladdr, physicaladdr;
    unsigned long *mmu_tlb_base = (unsigned long *)0x30000000;

    /*
     * Steppingstone的起始物理地址为0，第一部分程序的起始运行地址也是0，
     * 为了在开启MMU后仍能运行第一部分的程序，
     * 将0～1M的虚拟地址映射到同样的物理地址
     */
    virtuladdr = 0;
    physicaladdr = 0;
    *(mmu_tlb_base + (virtuladdr >> 20)) = (physicaladdr & 0xFFF00000) | \
                                            MMU_SECDESC_WB;

    /*
     * 0x56000000是GPIO寄存器的起始物理地址，
     * GPBCON和GPBDAT这两个寄存器的物理地址0x56000050、0x56000054，
     * 为了在第二部分程序中能以地址0xA0000050、0xA0000054来操作GPFCON、GPFDAT，
     * 把从0xA0000000开始的1M虚拟地址空间映射到从0x56000000开始的1M物理地址空间
     */
    virtuladdr = 0xA0000000;
    physicaladdr = 0x56000000;
    *(mmu_tlb_base + (virtuladdr >> 20)) = (physicaladdr & 0xFFF00000) | \
                                            MMU_SECDESC;

    /*
     * SDRAM的物理地址范围是0x30000000～0x33FFFFFF，
     * 将虚拟地址0xB0000000～0xB3FFFFFF映射到物理地址0x30000000～0x33FFFFFF上，
     * 总共64M，涉及64个段描述符
     */
    virtuladdr = 0xB0000000;
    physicaladdr = 0x30000000;
    while (virtuladdr < 0xB4000000)
    {
        *(mmu_tlb_base + (virtuladdr >> 20)) = (physicaladdr & 0xFFF00000) | \
                                                MMU_SECDESC_WB;
        virtuladdr += 0x100000;
        physicaladdr += 0x100000;
    }
}

/*
 * 启动MMU
 */
void mmu_init(void)
{
    unsigned long ttb = 0x30000000;

__asm__(
    "mov    r0, #0\n"
    "mcr    p15, 0, r0, c7, c7, 0\n"    /* 使无效ICaches和DCaches */

    "mcr    p15, 0, r0, c7, c10, 4\n"   /* drain write buffer on v4 */
    "mcr    p15, 0, r0, c8, c7, 0\n"    /* 使无效指令、数据TLB */

    "mov    r4, %0\n"                   /* r4 = 页表基址 */
    "mcr    p15, 0, r4, c2, c0, 0\n"    /* 设置页表基址寄存器 */

    "mvn    r0, #0\n"                   
    "mcr    p15, 0, r0, c3, c0, 0\n"    /* 域访问控制寄存器设为0xFFFFFFFF，
                                         * 不进行权限检查 
                                         */    
    /* 
     * 对于控制寄存器，先读出其值，在这基础上修改感兴趣的位，
     * 然后再写入
     */
    "mrc    p15, 0, r0, c1, c0, 0\n"    /* 读出控制寄存器的值 */

    /* 控制寄存器的低16位含义为：.RVI ..RS B... .CAM
     * R : 表示换出Cache中的条目时使用的算法，
     *     0 = Random replacement；1 = Round robin replacement
     * V : 表示异常向量表所在的位置，
     *     0 = Low addresses = 0x00000000；1 = High addresses = 0xFFFF0000
     * I : 0 = 关闭ICaches；1 = 开启ICaches
     * R、S : 用来与页表中的描述符一起确定内存的访问权限
     * B : 0 = CPU为小字节序；1 = CPU为大字节序
     * C : 0 = 关闭DCaches；1 = 开启DCaches
     * A : 0 = 数据访问时不进行地址对齐检查；1 = 数据访问时进行地址对齐检查
     * M : 0 = 关闭MMU；1 = 开启MMU
     */

    /*  
     * 先清除不需要的位，往下若需要则重新设置它们    
     */
                                        /* .RVI ..RS B... .CAM */ 
    "bic    r0, r0, #0x3000\n"          /* ..11 .... .... .... 清除V、I位 */
    "bic    r0, r0, #0x0300\n"          /* .... ..11 .... .... 清除R、S位 */
    "bic    r0, r0, #0x0087\n"          /* .... .... 1... .111 清除B/C/A/M */

    /*
     * 设置需要的位
     */
    "orr    r0, r0, #0x0002\n"          /* .... .... .... ..1. 开启对齐检查 */
    "orr    r0, r0, #0x0004\n"          /* .... .... .... .1.. 开启DCaches */
    "orr    r0, r0, #0x1000\n"          /* ...1 .... .... .... 开启ICaches */
    "orr    r0, r0, #0x0001\n"          /* .... .... .... ...1 使能MMU */

    "mcr    p15, 0, r0, c1, c0, 0\n"    /* 将修改的值写入控制寄存器 */
    : /* 无输出 */
    : "r" (ttb) );
}

leds.c

/*
 * leds.c: 循环点亮4个LED
 * 属于第二部分程序，此时MMU已开启，使用虚拟地址
 */ 

#define GPFCON      (*(volatile unsigned long *)0xA0000050)     // 物理地址0x56000050
#define GPFDAT      (*(volatile unsigned long *)0xA0000054)     // 物理地址0x56000054

#define    GPF4_out    (1<<(4*2))
#define    GPF5_out    (1<<(5*2))
#define    GPF6_out    (1<<(6*2))

/*
 * wait函数加上“static inline”是有原因的，
 * 这样可以使得编译leds.c时，wait嵌入main中，编译结果中只有main一个函数。
 * 于是在连接时，main函数的地址就是由连接文件指定的运行时装载地址。
 * 而连接文件mmu.lds中，指定了leds.o的运行时装载地址为0xB4004000，
 * 这样，head.S中的“ldr pc, =0xB4004000”就是跳去执行main函数。
 */
static inline void wait(volatile unsigned long dly)
{
    for(; dly > 0; dly--);
}

int main(void)
{
    unsigned long i = 0;

    GPFCON = GPF4_out|GPF5_out|GPF6_out;        // 将LED1,2,4对应的GPF4/5/6三个引脚设为输出

    while(1){
        wait(30000);
        GPFDAT = (~(i<<4));         // 根据i的值，点亮LED1,2,4
        if(++i == 8)
            i = 0;
    }

    return 0;
}

Android应用开发之点击事件的四种写法

第一种方法

●先在布局里面增加一个按钮控件

<Button
    android:id="@+id/btn1"
    android:layout_width="wrap_content"
    android:layout_height="wrap_content"
    android:layout_centerHorizontal="true"
    android:layout_centerVertical="true"
    android:text="点击" />

注意:上面按钮控件的id中，@+id表示创建一个控件，@id表示引用一个控件

●然后在MainActive.java中的onCreate方法中操作

1、获取控件

Button botton = (Button) findViewById(R.id.btn1);

findViewByid()返回的是一个控件，但是我希望使用按钮控件来定义，本来按钮这个类就是控件的子类，这里注意要强制转换一下，传入参数是这个控件的id，值得一提的是很多控件的id都在R这个类的id类下。

2、设置控件的点击事件

botton.setOnClickListener(new OnClickListener(){

        @Override
        public void onClick(View arg0) {
            //点击事件的内容
        }

    });

直接调用Button的方法即可，其参数是一个接口类型，在本例子中选择的是new一个局部内部类，名称就是这个接口的名称，然后覆盖掉里面的方法

3、设置点击事件的内容

public void onClick(View arg0) {

    System.out.println("zhuyanlin");

}

比如上面的System.out.println()语句就是一个内容，在这里的操作将会在点击事件触发时生效，即按下按钮的时候。

第二种方法

●增加按钮控件方式和第一种方法一致

●除了设置控件点击事件，其他与第一种方法一致

S3C2440嵌入式之SDRAM

发表于 2017-09-22 | 分类于 嵌入式linux开发

代码

head.s

.equ        sdram_base,        0x30000000  @仅仅是宏定义而已
.equ        mem_ctrl_reg,    0x48000000    

@;////////////////////////////////////////////////////////////////////////////////////////////    
.text
.global _start
_start:
    bl close_watch_dog
    bl set_memort_control_register
    bl steppingstone_copy_sdram
    ldr pc,=sdram_on
sdram_on:                                @如果这样ldr pc,=sdram_base，则cpu会一直从0x30000000运行，陷入死循环
    ldr sp,=0x34000000                    @一定别忘记设置栈
    bl main
main_loop:
    b main_loop

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @1、关闭看门狗
close_watch_dog:
    ldr r0,=0x53000000
    mov r1,#0x0000
    str r1,[r0]
    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @2、设置13个存储控制器，配置从外部SRAM中启动
set_memort_control_register:
    ldr r0,=mem_ctrl_reg
    adrl r1,memory_control_register_value
    add r3,r0,#52

loop1:
    ldr r2,[r1],#4
    str    r2,[r0],#4
    cmp r0,r3
    bne    loop1

    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
                                        @3、将steppingstone中的代码移动到外部SRAM中
steppingstone_copy_sdram:
    mov r1,#0
    ldr r2,=sdram_base
    add r3,r1,#1024*4
loop2:
    ldr r4,[r1],#4
    str r4,[r2],#4
    cmp r1,r3
    bne loop2

    mov pc,lr

@;////////////////////////////////////////////////////////////////////////////////////////////    
    @存储控制器的13个寄存器的值，这里顺便把其他外部设备如100M网卡DM9000等也一并配置了
.align 4   @好像少了这个是不行的
memory_control_register_value:
    .long    0x22011110    @BWSCON
    .long    0x00000700    @BANKCON0
    .long    0x00000700    @BANKCON1
    .long    0x00000700    @BANKCON2
    .long    0x00000700    @BANKCON3
    .long    0x00000700    @BANKCON4
    .long    0x00000700    @BANKCON5
    .long    0x00018005    @BANKCON6
    .long    0x00018005    @BANKCON7
    .long    0x008C07A3    @REFRESH
    .long    0x000000B1    @BANKSIZE
    .long    0x00000030    @MRAR6
    .long    0x00000030    @MRAR7

分析：这个汇编源文件主要是设置使用外部SDRAM运行程序。主要经过下面几步

1、关闭看门狗

2、设置位于0x48000000地址的13个存储控制器的值，使之支持SDRAM

由于这几个寄存器都是32位的，所以都是4个字节对齐了，严谨点要强制使用4个字节对齐：

.align 4

尤其注意：当获取13个控制寄存器的绝对地址时，直接使用ldr获取之前的宏定义即可，但是获取代码最后定义的13个寄存器的值的地址时，不能用ldr获取绝对地址。当时在链接时，我们指定代码存在的地址是外部SDRAM，即0x30000000开始的，但是目前程序还是在内部SRAM中运行，所以当前代码都处在“错误”的绝对地址，但是其相对偏移地址是正确的（自行脑补），所以此处一定要使用addl指令跳转至相对偏移地址。我之前就是使用ldr跳转的，当查看反汇编代码时，发现取不到寄存器的值，pc指针居然跳到了最后一个，即第13个寄存器的值的下一个位置，具体是怎么回事我也没有仔细研究了。

ldr r0,=mem_ctrl_reg
adrl r1,memory_control_register_value

3、将SteppingStone中的4KB代码复制到外部SDRAM

在一开始开发板是不带操作系统的并且是从Nand Flash启动，在CPU一上电启动，硬件会自动将Nand Flash中前4KB的代码复制到CPU中一个叫做SteppingStone中去，其本质就是内部SRAM中的全部4KB的空间，然后程序从内部SRAM中执行。仅仅使用一个简单的循环即可。

cmp和bne：在下面的语句中，先比较r1和r3的值是否相等，然后操作相应标志位，bne检查相应标志位，判断从而实现跳转或者退出循环

cmp的详解可以参考
http://blog.csdn.net/ftx2540993425/article/details/29926075

bne的详解可以参考
http://www.cnblogs.com/HangZhe/p/6916502.html

cmp r1,r3
bne loop2

4、跳转至外部SDRAM中继续执行代码

在程序运行到下面这步之前，其实际上还是在内部SRAM中执行代码，由于之前已经将外部SDRAM初始化了，所以现在需要让程序回到其应该运行的地方继续执行代码，即让pc指向SDRAM的某个地址。

注意：在第2条中说到过程序代码排列和链接其实地址有关，所有代码的地址都是（0x30000000+相对偏移地址），而现在pc仍在内部SRAM中，所以下一步必须跳转到绝对地址（正确的地址）运行程序。

ldr pc,=sdram_on

sdram_on标号的作用：我曾经使用“ldr pc,=0x30000000”直接跳到外部SDRAM的其实地址，发现程序死机了，因为它一执行到“ldr pc,=0x30000000”就会进行跳转到开头又重新运行，所以并不是随便跳的，一定要跳到在在外部SDRAM中和当前内部SRAM中即将执行一致的代码
处继续执行，即可转移到外部SDRAM中执行，不然还是在内部SRAM中执行，所以上面的跳转非常重要。

5、设置栈顶指针

栈顶指针设置的必要性就不阐述了，不然局部变量和函数跳转的寄存器的值放哪~

6、跳至main函数

一般是跳转到C源代码中的main函数了，到此为止，使用外部SDRAM的配置已全部完成，可以在main函数中尽情地编写代码了。
